    def compute_I_STI(self) -> Tuple[float, Dict]:
        """
        è®¡ç®—å°ºåº¦ä¸å˜æŒ‡æ•°ï¼ˆå…±è¯†A2ï¼‰
        
        è¿”å›:
            I_STI: å°ºåº¦ä¸å˜æŒ‡æ•°
            details: è¯¦ç»†ä¿¡æ¯å­—å…¸
        """
        O_values = []
        scale_details = []
        
        for scale in self.scales:
            O, detail = self.compute_O_freq(scale=scale)
            O_values.append(O)
            scale_details.append(detail)
        
        # è®¡ç®—å˜å¼‚ç³»æ•°
        O_mean = np.mean(O_values)
        O_std = np.std(O_values)
        CV = O_std / O_mean if O_mean > 0 else np.inf
        
        # I_STI
        I_STI = max(0, 1 - CV)
        
        # Bootstrapç½®ä¿¡åŒºé—´
        def cv_statistic(x):
            return 1 - (np.std(x) / np.mean(x)) if np.mean(x) > 0 else 0
        
        try:
            boot_result = bootstrap(
                (O_values,),
                cv_statistic,
                confidence_level=self.confidence,
                n_resamples=1000,
                random_state=42
            )
            ci_low = boot_result.confidence_interval.low
            ci_high = boot_result.confidence_interval.high
        except Exception as e:
            warnings.warn(f"Bootstrapå¤±è´¥: {e}")
            ci_low, ci_high = I_STI, I_STI
        
        # ç™½å™ªå£°åŸºå‡†å¯¹æ¯”ï¼ˆå®šç†B.1ï¼‰
        M = len(self.scales)
        L = len(self.signal)
        I_STI_noise = 1 - np.sqrt(np.pi / (2 * M)) / np.sqrt(L)
        
        # ç»Ÿè®¡æ˜¾è‘—æ€§ï¼ˆvsç™½å™ªå£°ï¼‰
        from scipy.stats import mannwhitneyu
        noise_samples = [I_STI_noise + np.random.normal(0, 0.05) for _ in range(100)]
        _, p_value = mannwhitneyu([I_STI], noise_samples, alternative='greater')
        
        return I_STI, {
            'O_per_scale': O_values,
            'O_mean': O_mean,
            'O_std': O_std,
            'CV': CV,
            'confidence_interval': (ci_low, ci_high),
            'I_STI_noise_baseline': I_STI_noise,
            'p_value_vs_noise': p_value,
            'scale_details': scale_details
        }
    
    def compute_energy_spectrum(self) -> Tuple[np.ndarray, np.ndarray]:
        """
        è®¡ç®—èƒ½é‡å¯†åº¦è°± e(f)ï¼ˆå…±è¯†A3ï¼‰

    def compute_thermodynamic_entropy(self) -> Tuple[float, Dict]:
        """
        è®¡ç®—çƒ­åŠ›å­¦ç†µï¼ˆå…±è¯†B1ï¼Œå®éªŒæ€§ï¼‰
        
        S = -k_B * Î£ P(f_i) * log(P(f_i)) * E(f_i) / (k_B * T)
        
        è¿”å›:
            S_thermo: çƒ­åŠ›å­¦ç†µ(J/K)
            details: è¯¦ç»†ä¿¡æ¯
        """
        freqs, e_f = self.compute_energy_spectrum()
        
        # å½’ä¸€åŒ–æ¦‚ç‡
        P = e_f / np.sum(e_f)
        
        # Shannonç†µéƒ¨åˆ†
        S_shannon = -np.sum(P * np.log(P + 1e-12))
        
        # æ€»èƒ½é‡
        E_total = np.sum(e_f)
        
        # çƒ­åŠ›å­¦ç†µï¼ˆæ ¹æ®å…±è¯†B1ï¼‰
        S_thermo = self.k_B * S_shannon * E_total / (self.k_B * self.T)
        
        # ä¸ç»å…¸çƒ­åŠ›å­¦å¯¹æ¯”ï¼ˆç†æƒ³æ°”ä½“ï¼‰
        # S_ideal_gas â‰ˆ k_B * N * (3/2 * ln(T) + const)
        # è¿™é‡Œä»…ä½œä¸ºæ•°é‡çº§å‚è€ƒ
        
        return S_thermo, {
            'S_shannon': S_shannon,
            'E_total': E_total,
            'temperature': self.T,
            'formula': 'S = k_B * S_shannon * E_total / (k_B * T)',
            'consensus_level': 'â­â­â­ (Bçº§å…±è¯†ï¼Œå®éªŒéªŒè¯ä¸è¶³)'
        }
    
    def compute_E_rate(self, historical_signals: List[np.ndarray]) -> Tuple[float, Dict]:
        """
        è®¡ç®—æ¶Œç°é€Ÿç‡ï¼ˆå…±è¯†B2ï¼Œéœ€å†å²æ•°æ®ï¼‰
        
        å‚æ•°:
            historical_signals: å†å²ä¿¡å·åˆ—è¡¨ï¼ˆæŒ‰æ—¶é—´é¡ºåºï¼‰
        
        è¿”å›:
            E_rate: æ¶Œç°é€Ÿç‡
            details: è¯¦ç»†ä¿¡æ¯
        """
        if len(historical_signals) < 2:
            warnings.warn("å†å²æ•°æ®ä¸è¶³ï¼ˆ<2ä¸ªæ—¶é—´ç‚¹ï¼‰ï¼ŒE_rateè®¾ä¸º0")
            return 0.0, {'error': 'insufficient_data', 'min_required': 2}
        
        # æå–æ¯ä¸ªæ—¶é—´ç‚¹çš„ä¸»é¢‘è°±
        all_peaks = []
        for sig in historical_signals:
            freqs, e_f = welch(sig, fs=self.fs)
            P = e_f / np.sum(e_f)
            threshold = 0.1 * np.max(P)
            peaks = set(freqs[P > threshold])
            all_peaks.append(peaks)
        
        # è®¡ç®—æ¯ä»£çš„æ–°å³°æ•°é‡
        new_peaks_per_generation = []
        for i in range(1, len(all_peaks)):
            new_peaks = all_peaks[i] - all_peaks[i-1]
            total_peaks = len(all_peaks[i])
            if total_peaks > 0:
                ratio = len(new_peaks) / total_peaks
                new_peaks_per_generation.append(ratio)
        
        # å¹³å‡æ¶Œç°é€Ÿç‡
        if new_peaks_per_generation:
            E_rate = np.mean(new_peaks_per_generation)
            E_std = np.std(new_peaks_per_generation)
        else:
            E_rate = 0.0
            E_std = 0.0
        
        return E_rate, {
            'new_peaks_per_gen': new_peaks_per_generation,
            'E_mean': E_rate,
            'E_std': E_std,
            'n_timepoints': len(historical_signals),
            'peak_history': [len(p) for p in all_peaks]
        }
