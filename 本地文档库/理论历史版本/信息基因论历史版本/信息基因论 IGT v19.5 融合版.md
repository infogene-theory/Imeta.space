def calculate_DTR_growth_rate(sensor_network, window_weeks=8):
    """
    è®¡ç®—DTRçš„å¢é•¿ç‡ï¼ˆè¿›åŒ–é€Ÿåº¦ï¼‰
    
    è¿”å›: DTRå¢é•¿ç‡ (per week)
    """
    if len(sensor_network.history) < window_weeks * 7:
        return None
    
    # è®¡ç®—å‰åŠæ®µå’ŒååŠæ®µçš„DTR
    mid_point = window_weeks * 7 // 2
    
    early_temps = [h['T_effective'] 
                   for h in sensor_network.history[:mid_point]]
    late_temps = [h['T_effective'] 
                  for h in sensor_network.history[mid_point:]]
    
    DTR_early = np.max(early_temps) - np.min(early_temps)
    DTR_late = np.max(late_temps) - np.min(late_temps)
    
    # å½’ä¸€åŒ–
    T0 = sensor_network.T0 or np.mean(early_temps + late_temps)
    
    DTR_growth = (DTR_late - DTR_early) / T0 / (window_weeks / 2)
    
    return DTR_growth
