def measure_order_complete(signal, sampling_rate):
    """
    完整的秩序度测量协议
    
    参数:
        signal: 时间序列数据
        sampling_rate: 采样率 (Hz)
    
    返回:
        results: 包含O_freq, CV, 判定结果的字典
    """
    # 步骤1: 预处理
    signal = preprocess(signal)
    # - 去趋势 (线性)
    # - 去除工频干扰 (50/60 Hz)
    # - 归一化 (零均值, 单位方差)
    # - 镜像填充 (处理边界效应)
    
    # 步骤2: 多尺度分析
    O_values = []
    scales = [2**i for i in range(1, 9)]  # [2, 4, 8, ..., 256]
    
    for scale in scales:
        # 2.1 傅里叶变换
        spectrum = np.fft.fft(signal[::scale])
        power = np.abs(spectrum)**2
        
        # 2.2 归一化功率谱
        P = power / np.sum(power)
        
        # 2.3 计算频谱熵
        S_freq = -np.sum(P * np.log2(P + 1e-12))
        
        # 2.4 计算秩序度
        O_scale = 1 - S_freq / np.log2(len(spectrum))
        O_values.append(O_scale)
    
    # 步骤3: 一致性检验
    mean_O = np.mean(O_values)
    std_O = np.std(O_values)
    CV = std_O / mean_O
    
    # 步骤4: 判定
    if CV < 0.2:
        verdict = "真秩序"
        confidence = 1 - CV / 0.2
    else:
        verdict = "假秩序"
        confidence = min(CV / 2.0, 1.0)
    
    return {
        'O_mean': mean_O,
        'O_std': std_O,
        'CV': CV,
        'O_per_scale': O_values,
        'verdict': verdict,
        'confidence': confidence
    }

F.2 RVS参数估计
def estimate_RVS_parameters(time_series_IG):
    """
    从历史数据估计RVS参数
    
    参数:
        time_series_IG: 信息基因数量的时间序列 [IG(t1), IG(t2), ...]
    
    返回:
        params: {r, d, mu} 参数字典
    """
    import numpy as np
    from scipy.optimize import curve_fit
    
    # 定义RVS模型
    def RVS_model(t, r, d, mu):
        """简化的RVS微分方程数值解"""
        IG = time_series_IG[0]
        dt = t[1] - t[0]
        result = [IG]
        
        for i in range(1, len(t)):
            dIG = (r - d) * IG + mu * np.random.normal(0, IG * 0.1)
            IG += dIG * dt
            result.append(IG)
        
        return np.array(result)
    
    # 拟合参数
    t = np.arange(len(time_series_IG))
    popt, pcov = curve_fit(RVS_model, t, time_series_IG, 
                           p0=[0.1, 0.05, 0.01],
                           bounds=([0, 0, 0], [1, 1, 1]))
    
    r, d, mu = popt
    
    return {
        'replication_rate': r,
        'death_rate': d,
        'mutation_rate': mu,
        'net_growth': r - d,
        'covariance': pcov
    }

        'app_AI': 'AI' in text and '意识' in text,
        'boundary': '不适用' in text or '不能解释' in text
    }
    
    # 合并相关检查
    core_checks = {
        '公理': checks['axiom'],
        'RVS机制': checks['RVS'],
        '多尺度验证': checks['multiscale'],
        '核心公式': checks['formula_O'] and checks['formula_RVS'],
        '三大应用': (checks['app_medical'] and 
                     checks['app_finance'] and 
                     checks['app_AI']),
        '边界声明': checks['boundary']
    }
    
    fidelity = sum(core_checks.values()) / len(core_checks)
    
    report = {
        'fidelity': fidelity,
        'status': 'PASS' if fidelity >= 0.9 else 'FAIL',
        'details': core_checks
    }
    
    return fidelity, report
