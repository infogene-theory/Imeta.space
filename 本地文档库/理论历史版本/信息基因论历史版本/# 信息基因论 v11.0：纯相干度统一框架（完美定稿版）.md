import numpy as np
from scipy.signal import welch, hilbert
from typing import Dict, Tuple

class IGTKernel:
    """
    IGT v11.0 计算内核：可模拟任意系统的相干演化
    
    核心功能：
    1. 实时计算C(t)及三维度分量
    2. 判定RVSE阶段
    3. 预测涌现时刻(d²Ω/dt² > 0)
    """
    
    def __init__(self, 
                 E_input: float,       # 能量输入率
                 C_env: float,         # 环境相干度
                 C_critical: float = 0.782):  # 涌现阈值
        self.E_input = E_input
        self.C_env = C_env
        self.C_critical = C_critical
        
        # RVSE参数
        self.k_r = 1.0    # 复制速率
        self.k_d = 0.5    # 淘汰速率
        self.k_mu = 0.2   # 变异速率
        self.gamma = 2.0  # 涌现增益
        
        # 历史记录
        self.history = {
            'C': [],
            'Omega': [],
            'dOmega_dt': [],
            'd2Omega_dt2': [],
            'stage': []
        }
    
    def compute_C_from_signal(self, signal: np.ndarray, fs: float) -> Tuple[float, Dict]:
        """
        从时间序列计算相干度（完整三维分解）
        
        参数:
            signal: 时间序列数据
            fs: 采样率(Hz)
        
        返回:
            C: 全局相干度
            details: 三维度详情
        """
        # 时间相干度（频域集中度）
        freqs, pxx = welch(signal, fs=fs, nperseg=min(256, len(signal)//4))
        C_temporal = np.max(pxx) / np.sum(pxx)
        
        # 空间相干度（多尺度相位一致性）
        scales = [2**i for i in range(1, 6)]
        phase_diffs = []
        for scale in scales:
            if len(signal) < scale * 2:
                continue
            n_segments = len(signal) // scale
            segments = signal[:n_segments*scale].reshape(n_segments, scale)
            analytic = hilbert(segments, axis=1)
            phases = np.angle(analytic)
            phase_diff = np.std(phases[:, scale//2])
            phase_diffs.append(phase_diff)
        C_spatial = 1 - np.mean(phase_diffs) / np.pi if phase_diffs else 0.5
        
        # 方向相干度（Hilbert相位向量）
        analytic_full = hilbert(signal)
        phases_full = np.angle(analytic_full)
        phase_vectors = np.column_stack([np.cos(phases_full), np.sin(phases_full)])
        norms = np.linalg.norm(phase_vectors, axis=1, keepdims=True)
        normalized = phase_vectors / (norms + 1e-12)
        net_vector = np.sum(normalized, axis=0)
        C_directional = np.linalg.norm(net_vector) / len(normalized)
        
        # 全局相干度（几何平均）
        C = (C_temporal * C_spatial * C_directional) ** (1/3)
        
        return C, {
            'C_temporal': C_temporal,
            'C_spatial': C_spatial,
            'C_directional': C_directional,
            'weights': (
                C_temporal / (C_temporal + C_spatial + C_directional),
                C_spatial / (C_temporal + C_spatial + C_directional),
                C_directional / (C_temporal + C_spatial + C_directional)
            )
        }
    
    def evolve_one_step(self, C_current: float, Omega_current: float, dt: float = 1.0) -> Dict:
        """
        演化一步（RVSE算法）
        
        返回:
            new_state: 包含新C, Omega, 阶段等信息
        """
        # 计算Omega的导数
        if len(self.history['Omega']) >= 2:
            dOmega_dt = (Omega_current - self.history['Omega'][-1]) / dt
            if len(self.history['dOmega_dt']) >= 1:
                d2Omega_dt2 = (dOmega_dt - self.history['dOmega_dt'][-1]) / dt
            else:
                d2Omega_dt2 = 0
        else:
            dOmega_dt = 0
            d2Omega_dt2 = 0
        
        # RVSE演化逻辑
        if C_current >= self.C_critical and d2Omega_dt2 > 0:
            # 涌现阶段
            stage = "Emergence"
            dC_dt = self.gamma * (C_current - self.C_critical) * (1 - C_current)
            dOmega_dt += self.gamma * C_current * self.E_input
            
        elif 0.6 <= C_current < self.C_critical:
            # 稳定阶段
            stage = "Stabilization"
            dC_dt = self.k_r * C_current**2 - self.k_d / (self.C_env + 1e-6)
            dOmega_dt += self.k_r * C_current * self.E_input
            
        elif 0.4 <= C_current < 0.6:
            # 复制阶段
            stage = "Replication"
            dC_dt = self.k_r * C_current - self.k_d / (self.C_env + 1e-6)
            dOmega_dt += 0.5 * self.E_input
            
        else:  # C < 0.4
            # 变异阶段
            stage = "Variation"
            dC_dt = -self.k_d / (self.C_env + 1e-6) + self.k_mu * (1 - C_current)
            dOmega_dt -= 0.3 * self.E_input
        
        # 更新状态
        C_new = max(0.0, min(1.0, C_current + dC_dt * dt))
        Omega_new = max(0.0, Omega_current + dOmega_dt * dt)
        
        # 记录历史
        self.history['C'].append(C_new)
        self.history['Omega'].append(Omega_new)
        self.history['dOmega_dt'].append(dOmega_dt)
        self.history['d2Omega_dt2'].append(d2Omega_dt2)
        self.history['stage'].append(stage)
        
        return {
            'C': C_new,
            'Omega': Omega_new,
            'dOmega_dt': dOmega_dt,
            'd2Omega_dt2': d2Omega_dt2,
            'stage': stage,
            'emergence_readiness': min(1.0, C_new / self.C_critical) if C_new < self.C_critical else 1.0
        }
    
    def predict_emergence_time(self, C_initial: float, Omega_initial: float, max_steps: int = 1000) -> int:
        """
        预测涌现发生时刻
        
        返回:
            步数（-1表示在max_steps内未涌现）
        """
        C = C_initial
        Omega = Omega_initial
        
        for step in range(max_steps):
            state = self.evolve_one_step(C, Omega)
            C = state['C']
            Omega = state['Omega']
            
            if state['stage'] == "Emergence":
                return step
        
        return -1  # 未涌现
