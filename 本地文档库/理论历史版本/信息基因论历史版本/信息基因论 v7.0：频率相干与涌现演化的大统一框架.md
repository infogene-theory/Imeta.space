        CV = O_std / O_mean if O_mean > 0 else np.inf
        I_STI = max(0, 1 - CV)
        
        # Jaccardç›¸ä¼¼åº¦(è·¨å°ºåº¦ä¸»é¢‘è°±ä¸€è‡´æ€§)
        jaccard_scores = []
        for i in range(len(H_spectra)):
            for j in range(i+1, len(H_spectra)):
                intersection = len(H_spectra[i] & H_spectra[j])
                union = len(H_spectra[i] | H_spectra[j])
                if union > 0:
                    jaccard_scores.append(intersection / union)
        
        jaccard_mean = np.mean(jaccard_scores) if jaccard_scores else 0
        
        details = {
            'O_per_scale': O_values,
            'O_mean': O_mean,
            'O_std': O_std,
            'CV': CV,
            'jaccard': jaccard_mean,
            'H_spectra': H_spectra
        }
        
        return I_STI, details
    
    def calculate_E_rate(self) -> Tuple[float, Dict]:
        """
        è®¡ç®—æ¶Œç°é€Ÿç‡ï¼ˆéœ€è¦å†å²æ•°æ®ï¼‰
        E_rate = d(å±‚çº§æ•°)/dt
        """
        if len(self.history) < 3:
            warnings.warn("å†å²æ•°æ®ä¸è¶³ï¼Œæ— æ³•è®¡ç®—E_rate")
            return 0.0, {'method': 'insufficient_data'}
        
        # æ–¹æ³•1: è·¨å°ºåº¦æ–°å³°å‡ºç°ç‡
        new_peaks_per_generation = []
        
        for i in range(1, len(self.history)):
            prev_signal = self.history[i-1]
            curr_signal = self.history[i]
            
            # æå–ä¸»é¢‘è°±
            _, psd_prev = welch(prev_signal, fs=self.fs)
            _, psd_curr = welch(curr_signal, fs=self.fs)
            
            P_prev = psd_prev / np.sum(psd_prev)
            P_curr = psd_curr / np.sum(psd_curr)
            
            # æ£€æµ‹æ–°å³°
            peaks_prev = set(np.where(P_prev > self.theta * np.max(P_prev))[0])
            peaks_curr = set(np.where(P_curr > self.theta * np.max(P_curr))[0])
            new_peaks = peaks_curr - peaks_prev
            
            new_peaks_per_generation.append(len(new_peaks))
        
        # E_rate = å¹³å‡æ¯ä»£æ–°å³°æ•° / æ€»å³°æ•°
        total_peaks = len(np.where(psd_curr / np.sum(psd_curr) > 
                                   self.theta * np.max(psd_curr / np.sum(psd_curr)))[0])
        E_rate = np.mean(new_peaks_per_generation) / max(total_peaks, 1)
        
        details = {
            'method': 'peak_emergence',
            'new_peaks_per_gen': new_peaks_per_generation,
            'avg_new_peaks': np.mean(new_peaks_per_generation),
            'total_peaks': total_peaks
        }
        
        return E_rate, details
    
    def full_analysis(self, confidence_level: float = 0.95) -> Dict:
        """
        æ‰§è¡Œå®Œæ•´çš„v7.0åˆ†æ
        è¿”å›: {O_freq, I_STI, E_rate, RVSE_stage, verdict}
        """
        # 1. è®¡ç®—æœ¬ä½“å±‚æŒ‡æ ‡
        I_STI, sti_details = self.calculate_I_STI()
        O_freq = sti_details['O_mean']
        
        # 2. è®¡ç®—æœºåˆ¶å±‚æŒ‡æ ‡
        E_rate, e_details = self.calculate_E_rate()
        
        # 3. Bootstrapç½®ä¿¡åŒºé—´
        O_resamples = bootstrap(
            (sti_details['O_per_scale'],), 
            np.mean, 
            confidence_level=confidence_level,
            n_resamples=1000,
            random_state=42
        )
        
        I_resamples = bootstrap(
            (sti_details['O_per_scale'],),
            lambda x: max(0, 1 - np.std(x)/np.mean(x)),
            confidence_level=confidence_level,
            n_resamples=1000,
            random_state=42
        )
        
        # 4. v7.0ä¸‰æŒ‡æ ‡åˆ¤å®š
        is_ordered = O_freq > 0.6
        is_stable = I_STI > 0.8
        is_true_order = is_ordered and is_stable
        is_evolvable = is_true_order and E_rate > 0
        is_emergent = is_evolvable and E_rate > 0.01
        
        # 5. RVSEé˜¶æ®µè¯†åˆ«
        if not is_ordered:
            rvse_stage = 'æ— åºæ€ï¼ˆæ·˜æ±°ä¸­ï¼‰'
        elif is_ordered and not is_stable:
            rvse_stage = 'ä¼ªç§©åºï¼ˆå˜å¼‚æ¢ç´¢ä¸­ï¼‰'
        elif is_stable and E_rate < 0.01:
            rvse_stage = 'ç¨³å®šæ€ï¼ˆå¤åˆ¶ç»´æŒä¸­ï¼‰'
        elif is_emergent:
            rvse_stage = 'æ¶Œç°æ€ï¼ˆå±‚çº§è·ƒè¿ä¸­ï¼‰'
        else:
            rvse_stage = 'è¿‡æ¸¡æ€'
        
        # 6. ç»Ÿè®¡æ£€éªŒ
        # ä¸ç™½å™ªå£°çš„æ˜¾è‘—æ€§æ£€éªŒ
        noise = np.random.randn(len(self.signal))
        O_noise = self.calculate_O_freq(noise)
        _, p_value = mannwhitneyu([O_freq], [O_noise], alternative='greater')
        
        return {
            # æœ¬ä½“å±‚
            'O_freq': O_freq,
            'O_freq_ci': (O_resamples.confidence_interval.low,
                         O_resamples.confidence_interval.high),
            'I_STI': I_STI,
            'I_STI_ci': (I_resamples.confidence_interval.low,
                        I_resamples.confidence_interval.high),
            'jaccard_similarity': sti_details['jaccard'],
            
            # æœºåˆ¶å±‚
            'E_rate': E_rate,
            'E_details': e_details,
            
            # åˆ¤å®š
            'is_true_order': is_true_order,
            'is_evolvable': is_evolvable,
            'is_emergent': is_emergent,
            'RVSE_stage': rvse_stage,
            
            # ç»Ÿè®¡
            'p_value_vs_noise': p_value,
            'confidence': confidence_level if is_true_order else 1 - I_STI,
            
            # åŸå§‹æ•°æ®
            'details': {**sti_details, **e_details}
        }
