### **5.3 熵涨落PID自动控制**

```python
class EntropyFluctuationController:
    """基于熵涨落的PID温控系统"""
    
    def __init__(self, target_delta_S, Kp=1.0, Ki=0.1, Kd=0.05):
        self.target = target_delta_S  # 目标熵涨落
        self.Kp, self.Ki, self.Kd = Kp, Ki, Kd
        self.integral = 0
        self.last_error = 0
    
    def control(self, current_delta_S, dt=1.0):
        """
        输入: 当前熵涨落 δS/⟨S⟩
        输出: 控制量（正=加热↑δS, 负=降温↓δS）
        """
        error = self.target - current_delta_S
        
        # P: 比例项（当前偏差）
        P = self.Kp * error
        
        # I: 积分项（历史累积）
        self.integral += error * dt
        I = self.Ki * self.integral
        
        # D: 微分项（变化趋势）
        D = self.Kd * (error - self.last_error) / dt
        
        self.last_error = error
        control_signal = P + I + D
        
        return {
            'total_control': control_signal,
            'P_term': P,
            'I_term': I,
            'D_term': D,
            'action': '加热' if control_signal > 0 else '降温',
            'magnitude': abs(control_signal)
        }
```

```python
class CriticalPointWarning:
    """临界点预警算法"""
    
    def __init__(self, window=100):
        self.window = window
        self.history = []
    
    def update(self, current_state):
        """更新状态历史"""
        self.history.append(current_state)
        if len(self.history) > self.window:
            self.history.pop(0)
    
    def detect_critical(self):
        """检测临界点信号"""
        if len(self.history) < 20:
            return {'warning': False, 'signals': []}
        
        signals = []
        recent = self.history[-20:]
        
        # 信号1：任一维度快速衰减
        for dim in ['I_S', 'I_omega', 'I_C']:
            values = [s[dim] for s in recent]
            trend = (values[-1] - values[0]) / values[0]
            if trend < -0.1:  # 10%下降
                signals.append(f"{dim}快速衰减: {trend:.2%}")
        
        # 信号2：三维比例剧烈波动
        ratios_history = []
        for s in recent:
            ratios = [
                s['I_omega'] / (s['I_S'] + 1e-10),
                s['I_C'] / (s['I_omega'] + 1e-10),
                s['I_S'] / (s['I_C'] + 1e-10)
            ]
            ratios_history.append(ratios)
        
        ratio_variance = np.var(ratios_history, axis=0).mean()
        if ratio_variance > 0.15:
            signals.append(f"三维比例失衡加剧: {ratio_variance:.3f}")
        
        # 信号3：总惯性非线性变化
        total_inertia = [s['I_S'] * s['I_omega'] * s['I_C'] 
                         for s in recent]
        second_deriv = np.gradient(np.gradient(total_inertia))
        if np.any(np.abs(second_deriv[-5:]) > 0.1):
            signals.append("总惯性出现拐点（可能发生相变）")
        
        return {
            'warning': len(signals) > 0,
            'signals': signals,
            'criticality_score': len(signals) / 3.0
        }
```
