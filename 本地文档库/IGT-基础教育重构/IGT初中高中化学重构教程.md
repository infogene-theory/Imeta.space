    """
    åŸå­æ ¸ç¨³å®šæ€§ï¼ˆé¢‘ç‡ç›¸å¹²åº¦ï¼‰
    Z: è´¨å­æ•°
    N: ä¸­å­æ•°
    """
    A = Z + N
    
    # è´¨å­é¢‘ç‡ï¼ˆå¸¦æ­£ç”µï¼‰
    proton_freq = 1.0 * Z
    
    # ä¸­å­é¢‘ç‡ï¼ˆä¸­æ€§ï¼Œç›¸ä½åç§»Ï€/2ï¼‰
    neutron_freq = 1.0 * N * np.exp(1j * np.pi/2)

    total_freq = proton_freq + neutron_freq
    
    # ç›¸å¹²åº¦ = |æ€»é¢‘ç‡|Â² / (è´¨å­é¢‘ç‡Â² + ä¸­å­é¢‘ç‡Â²)

    symmetry_factor = np.exp(-abs(Z-N)/A)
    
    # æœ€ç»ˆç¨³å®šæ€§
    stability = coherence * symmetry_factor
    
    return stability
